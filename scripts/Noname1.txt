SELECT DISTINCT(IVD.SUPPLIER_ID)
FROM LOT_ITEM LI
LEFT OUTER JOIN INVENTORY_DOC IVD ON (LI.INVENTORY_DOC_ID = IVD.INVENTORY_DOC_ID)

SELECT IVD.SUPPLIER_ID, SUM(LI.TX_AMOUNT), SUM(TOTAL_INCLUDE_PRICE)
FROM LOT_ITEM LI
LEFT OUTER JOIN INVENTORY_DOC IVD ON (LI.INVENTORY_DOC_ID = IVD.INVENTORY_DOC_ID)
GROUP BY IVD.SUPPLIER_ID

SELECT LI.PART_ITEM_ID, LI.LOCATION_ID, LI.LAYOUT_ID, SUM(LI.LEFT_AMOUNT) LEFT_AMOUNT
FROM LOT_ITEM LI
GROUP BY LI.PART_ITEM_ID, LI.LOCATION_ID, LI.LAYOUT_ID
